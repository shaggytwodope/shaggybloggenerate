cached() {
    exists=1
    while read line; do
        linea=(${line// / })
        sha=`sha1sum $1`
        sha=(${sha// / })
        if [ ${linea[1]} == $1 ]
        then
            if [ ${linea[0]} == ${sha[0]} ]; then
                exists=0
                break
            else
                sed -i "\|^$line\$|d" $ABSPATH/.cache
            fi
        fi
    done < $ABSPATH/.cache

    if [ $exists -eq 1 ]; then
        echo `sha1sum $1` >> $ABSPATH/.cache
    return 1
else
    return 0
fi

}
